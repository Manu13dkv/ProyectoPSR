51NS_LOG_COMPONENT_DEFINE("Main");
52
53int main(int argc, char* argv[]){
54
55
56 GlobalValue::Bind ("SimulatorImplementationType", StringValue ("ns3::RealtimeSimulatorImpl"));
57 GlobalValue::Bind ("ChecksumEnabled", BooleanValue (true));
58
59 // Default Values.
60 //
61 // *_r (Raw) sufix to advise that the variable will be wrapped
62 // by an ns3 object, such as Time object to a double.
63 //
64 //================================================================================
65 std::string sim_len_r = "3600s";
66 uint32_t n_nodes = 2;
67 std::string base_ip_r = "10.1.1.0";
68 std::string base_mask_r = "255.255.255.0";
69 //================================================================================
70
71 CommandLine cmd(__FILE__);
72
73 cmd.AddValue("sim_len", "Length of the simulation.", sim_len_r);
74 cmd.AddValue("n_nodes", "Number of real nodes to insert in the simulation.", n_nodes);
75
76 cmd.Parse(argc, argv);
77
78 // Wrapping parameters.
79 //
80 //================================================================================
81 Time sim_len = Time(sim_len_r);
82 Ipv4Address base_ip = Ipv4Address(base_ip_r.c_str());
83 Ipv4Mask base_mask = Ipv4Mask(base_mask_r.c_str());
84 //================================================================================
85
86 // Configuring scenario.
87 //================================================================================
88
89 Ipv4AddressHelper ip_helper;
90 ip_helper.SetBase(base_ip, base_mask);
91
92 Ptr<Ipv4Router> router = CreateObject<Ipv4Router>();
93 uint32_t tap_index = 1;
94
95 // Leave free eth0 to connect the management device.
96 for ( uint32_t node = 1; node <= n_nodes; node++){
97
98 router->IfUp(
99 "eth"+std::to_string(node), // --> eth1, eth2 ...
100 ip_helper.NewAddress(),
101 base_mask
102 );
103
104 Ptr<EmuAdapter> tap = CreateObject<EmuAdapter>();
105
106 Ptr<NetDevice> guest_iface = router->Link(
107 tap->GetNode(),
108 "eth"+std::to_string(node),
109 ip_helper.NewAddress(),
110 base_mask
111 );
112
113 // When ns3 creates a tap device, then our system links it to a bridge
114 // and this to another tap device, hence if ns3 create tap1, our shell script will
115 // generate tap 2. Thus, the next tap in ns3 will be directly tap3.
116 tap->AttachTapDevice("tap"+std::to_string(tap_index), guest_iface);
117 tap_index+=2;
118 ip_helper.NewNetwork();
119 }
120
121 // Management device.
122 // Feel free to use it to manage the system from the host.
123 Ptr<EmuAdapter> management_tap = CreateObject<EmuAdapter>();
124
125 router->IfUp(
126 "eth0",
127 Ipv4Address("10.1.10.1"),
128 Ipv4Mask("255.255.255.0")
129 );
130
131 Ptr<NetDevice> management_iface = router->Link(
132 management_tap->GetNode(),
133 "eth0",
134 Ipv4Address("10.1.10.2"),
135 Ipv4Mask("255.255.255.0")
136 );
137
138 management_tap->AttachTapDevice("tap0", management_iface);
139
140 // Global Config
141 //================================================================================
142 router->EnablePcap();
143 Ipv4GlobalRoutingHelper::PopulateRoutingTables ();
144
145
146 // Simulation
147 //================================================================================
148 NS_LOG_INFO ("Run Emulation.");
149 Simulator::Stop (sim_len);
150 Simulator::Run ();
151 Simulator::Destroy ();
152 NS_LOG_INFO ("Done.");
153
154}
